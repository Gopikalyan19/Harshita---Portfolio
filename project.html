Project Overview

The EMG Monitoring System is a comprehensive solution designed to measure and analyze electromyographic signals for medical diagnostics and rehabilitation purposes. This project addresses the need for affordable, portable EMG monitoring devices in healthcare settings.

The system captures muscle electrical activity through surface electrodes, processes the signals in real-time, and provides meaningful data visualization for healthcare professionals. It features wireless connectivity, real-time monitoring, and customizable alert systems.

Technical Implementation

emg_signal_processing.cpp
#include <Arduino.h>
#include <WiFi.h>
#include <WebSocketsServer.h>
#include <ArduinoJson.h>

class EMGProcessor {
private:
    static const int SAMPLE_RATE = 1000;
    static const int BUFFER_SIZE = 100;
    static const int EMG_PIN = A0;
    
    float signalBuffer[BUFFER_SIZE];
    int bufferIndex = 0;
    unsigned long lastSampleTime = 0;
    
    // Filter coefficients for 20-450Hz bandpass filter
    float filterCoeffs[5] = {0.0201, 0.0804, 0.1206, 0.0804, 0.0201};
    float filterHistory[5] = {0, 0, 0, 0, 0};
    
public:
    void initialize() {
        pinMode(EMG_PIN, INPUT);
        Serial.begin(115200);
        setupWiFi();
        Serial.println("EMG Processor initialized");
    }
    
    float readEMGSample() {
        int rawValue = analogRead(EMG_PIN);
        float voltage = (rawValue * 3.3) / 4095.0; // Convert to voltage
        
        // Apply high-pass filter to remove DC offset
        float filteredSignal = applyBandpassFilter(voltage);
        
        // Rectify the signal
        float rectifiedSignal = abs(filteredSignal);
        
        // Apply smoothing filter
        float smoothedSignal = applySmoothingFilter(rectifiedSignal);
        
        return smoothedSignal;
    }
    
    float applyBandpassFilter(float input) {
        // Shift filter history
        for (int i = 4; i > 0; i--) {
            filterHistory[i] = filterHistory[i-1];
        }
        filterHistory[0] = input;
        
        // Apply filter
        float output = 0;
        for (int i = 0; i < 5; i++) {
            output += filterCoeffs[i] * filterHistory[i];
        }
        
        return output;
    }
    
    float applySmoothingFilter(float input) {
        // Simple moving average
        signalBuffer[bufferIndex] = input;
        bufferIndex = (bufferIndex + 1) % BUFFER_SIZE;
        
        float sum = 0;
        for (int i = 0; i < BUFFER_SIZE; i++) {
            sum += signalBuffer[i];
        }
        
        return sum / BUFFER_SIZE;
    }
    
    void processSignal() {
        unsigned long currentTime = millis();
        
        if (currentTime - lastSampleTime >= (1000 / SAMPLE_RATE)) {
            float emgValue = readEMGSample();
            
            // Create JSON data packet
            StaticJsonDocument<200> doc;
            doc["timestamp"] = currentTime;
            doc["emg_value"] = emgValue;
            doc["muscle_activity"] = calculateMuscleActivity(emgValue);
            doc["alert_level"] = checkAlertLevel(emgValue);
            
            String jsonString;
            serializeJson(doc, jsonString);
            
            // Send via WebSocket
            broadcastEMGData(jsonString);
            
            lastSampleTime = currentTime;
        }
    }
    
    float calculateMuscleActivity(float emgValue) {
        // Convert EMG voltage to percentage of maximum voluntary contraction
        float maxEMG = 3.0; // Calibrated maximum value
        return (emgValue / maxEMG) * 100.0;
    }
    
    String checkAlertLevel(float emgValue) {
        if (emgValue < 0.1) return "low";
        else if (emgValue < 0.5) return "normal";
        else if (emgValue < 1.0) return "high";
        else return "critical";
    }
    
    void setupWiFi() {
        WiFi.begin("your_wifi_ssid", "your_wifi_password");
        while (WiFi.status() != WL_CONNECTED) {
            delay(1000);
            Serial.println("Connecting to WiFi...");
        }
        Serial.println("WiFi connected");
        Serial.print("IP address: ");
        Serial.println(WiFi.localIP());
    }
    
    void broadcastEMGData(String data) {
        // Implementation for WebSocket broadcasting
        Serial.println("EMG Data: " + data);
    }
};

EMGProcessor emgProcessor;

void setup() {
    emgProcessor.initialize();
}

void loop() {
    emgProcessor.processSignal();
    delay(1); // Small delay to prevent overwhelming the processor
}

Challenges & Solutions

Challenge: Signal Noise Reduction

EMG signals are inherently noisy due to electrical interference and motion artifacts.

Solution:

Implemented multi-stage filtering including high-pass, low-pass, and notch filters to eliminate 60Hz interference and DC offset while preserving the EMG signal integrity.

Challenge: Real-time Processing

Processing EMG signals in real-time while maintaining accuracy and low latency.

Solution:

Optimized algorithms using efficient data structures and implemented circular buffers for continuous signal processing without memory overflow.
